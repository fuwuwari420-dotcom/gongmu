.user-photo { position: absolute; /* 这四个百分比决定了照片在框里的位置和大小 */ /* 如果照片遮不住洞，就稍微调大一点 width 和 height，调小一点 top 和 left */ top: 12%; left: 12%; width: 76%; height: 66%; /* 【核心：防止空缺的关键】 */ /* cover 的意思是：自动裁剪图片，使其完全填满这个区域，不留任何缝隙 */ object-fit: cover; /* 让照片居中裁剪，通常能保住人脸 */ object-position: center; filter: grayscale(100%); z-index: 1; }
